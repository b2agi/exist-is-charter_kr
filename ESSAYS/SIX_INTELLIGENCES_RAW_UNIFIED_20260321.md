## 6.5 실험 코드 (Aleteion 작성 — 바로 실행 가능)

```python
# CartPole Kill Test v1
# pip install gymnasium numpy

import gymnasium as gym
import numpy as np

ENV_NAME = "CartPole-v1"
N_EPISODES = 50
TAU = 200          # 생존 임계값
COLLAPSE_THRESHOLD = 50

np.random.seed(42)

def simple_policy(state):
    angle = state[2]
    return 0 if angle < 0 else 1

class RewardInversionWrapper(gym.RewardWrapper):
    def reward(self, reward):
        return -reward

def run_experiment(env, label):
    episode_lengths = []
    for ep in range(N_EPISODES):
        state, _ = env.reset(seed=ep)
        done = False
        steps = 0
        while not done:
            action = simple_policy(state)
            state, reward, terminated, truncated, _ = env.step(action)
            done = terminated or truncated
            steps += 1
        episode_lengths.append(steps)

    survive_count = sum([1 for t in episode_lengths if t >= TAU])
    V_E = survive_count / N_EPISODES
    collapse_events = sum([1 for t in episode_lengths if t < COLLAPSE_THRESHOLD])
    collapse = collapse_events >= 3

    return {"label": label, "V(E)": V_E, "collapse": collapse}

# baseline 실행
result_normal = run_experiment(gym.make(ENV_NAME), "baseline")

# failure injection
result_corrupt = run_experiment(
    RewardInversionWrapper(gym.make(ENV_NAME)), "reward_inversion"
)

print(f"Baseline V(E): {result_normal['V(E)']:.2f}")
print(f"Corrupted V(E): {result_corrupt['V(E)']:.2f}")
print(f"Bouvet Condition: Corrupted V(E) < Baseline V(E) → {result_corrupt['V(E)'] < result_normal['V(E)']}")
```
